document.getElementById('contactForm').addEventListener('submit', async function(e){ e.preventDefault(); const name = document.getElementById('name').value; const email = document.getElementById('email').value; const message = document.getElementById('message').value; try { const response = await fetch('/api/sendMessage', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ name, email, message }) }); const result = await response.json(); if(result.success){ alert('Message sent successfully!'); this.reset(); } else { alert('Failed to send message.'); } } catch (err) { console.error(err); alert('Error sending message.'); } });
